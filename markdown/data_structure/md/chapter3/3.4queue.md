#### EnQueue
```
Status EnQueue(LInkQueue &Q , QElemType e)
{
    p=(QueuePtr)malloc(sizeof(QNode));
    if(!p) exit (OVERFLOW);
    p->data=e;
    p->next=NULL;
    Q.rear->next=p;
    Q.rear=p;
    return OK;
}
```
#### DeQueue
```
Status DeQueue(LinkQueue& Q,QElemType &e)
{
    QueuePtr p;
    QElemType e;

    if (Q.front==Q.rear) return ERROR;
    p=Q.front->next;
    e=p->data;
    Q.front->next=p->next;
    if (Q.rear==p)
    Q.rear=Q.front;
    free(p);
    return OK;
}
